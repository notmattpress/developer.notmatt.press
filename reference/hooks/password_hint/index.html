add_filter( 'password_hint', 'change_password_hint_message' );
function change_password_hint_message( $hint ) {
    $hint = __( 'Hint: To make it stronger, use upper and lower case letters, numbers, and symbols like ! " ? $ % ^ & ).', 'text-domain' );

    return $hint;
}
