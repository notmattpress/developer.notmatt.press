wp_basename( string $path, string $suffix = '' ): string

Parameters

Return

Source

function wp_basename( $path, $suffix = '' ) {
	return urldecode( basename( str_replace( array( '%2F', '%5C' ), '/', urlencode( $path ) ), $suffix ) );
}
