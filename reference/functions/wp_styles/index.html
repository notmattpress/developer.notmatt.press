wp_styles()

wp_styles(): WP_Styles

Initializes $wp_styles if it has not been set.

Return

WP_Styles WP_Styles instance.

Source

function wp_styles() {
	global $wp_styles;

	if ( ! ( $wp_styles instanceof WP_Styles ) ) {
		$wp_styles = new WP_Styles();
	}

	return $wp_styles;
}
