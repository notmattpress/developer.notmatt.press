get_super_admins()

get_super_admins(): string[]

Retrieves a list of super admins.

Return

string[] List of super admin logins.

Source

function get_super_admins() {
	global $super_admins;

	if ( isset( $super_admins ) ) {
		return $super_admins;
	} else {
		return get_site_option( 'site_admins', array( 'admin' ) );
	}
}
