WP_Error::get_error_code()

WP_Error::get_error_code(): string|int

Retrieves the first error code available.

Return

string|int Empty string, if no error codes.

Source

public function get_error_code() {
	$codes = $this->get_error_codes();

	if ( empty( $codes ) ) {
		return '';
	}

	return $codes[0];
}
