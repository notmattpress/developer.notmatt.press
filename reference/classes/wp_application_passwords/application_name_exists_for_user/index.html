public static function application_name_exists_for_user( $user_id, $name ) {
	$passwords = static::get_user_application_passwords( $user_id );

	foreach ( $passwords as $password ) {
		if ( strtolower( $password['name'] ) === strtolower( $name ) ) {
			return true;
		}
	}

	return false;
}
