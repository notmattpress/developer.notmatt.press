WP_Application_Passwords::chunk_password()

Source

public static function chunk_password( $raw_password ) {
	$raw_password = preg_replace( '/[^a-z\d]/i', '', $raw_password );

	return trim( chunk_split( $raw_password, 4, ' ' ) );
}
