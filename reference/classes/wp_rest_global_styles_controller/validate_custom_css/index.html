Source

protected function validate_custom_css( $css ) {
	if ( preg_match( '#</?\w+#', $css ) ) {
		return new WP_Error(
			'rest_custom_css_illegal_markup',
			__( 'Markup is not allowed in CSS.' ),
			array( 'status' => 400 )
		);
	}
	return true;
}
