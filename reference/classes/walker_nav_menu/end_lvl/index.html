public function end_lvl( &$output, $depth = 0, $args = null ) {
	if ( isset( $args->item_spacing ) && 'discard' === $args->item_spacing ) {
		$t = '';
		$n = '';
	} else {
		$t = "\t";
		$n = "\n";
	}
	$indent  = str_repeat( $t, $depth );
	$output .= "$indent</ul>{$n}";
}
