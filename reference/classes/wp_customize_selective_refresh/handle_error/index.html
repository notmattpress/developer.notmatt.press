public function handle_error( $errno, $errstr, $errfile = null, $errline = null ) {
	$this->triggered_errors[] = array(
		'partial'      => $this->current_partial_id,
		'error_number' => $errno,
		'error_string' => $errstr,
		'error_file'   => $errfile,
		'error_line'   => $errline,
	);
	return true;
}
