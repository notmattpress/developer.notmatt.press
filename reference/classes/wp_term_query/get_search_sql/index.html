protected function get_search_sql( $search ) {
	global $wpdb;

	$like = '%' . $wpdb->esc_like( $search ) . '%';

	return $wpdb->prepare( '((t.name LIKE %s) OR (t.slug LIKE %s))', $like, $like );
}
