Source

protected static function get_plugins() {
	if ( is_array( self::$plugins ) ) {
		return self::$plugins;
	}

	require_once ABSPATH . '/wp-admin/includes/plugin.php';
	self::$plugins = get_plugins();

	return self::$plugins;
}
