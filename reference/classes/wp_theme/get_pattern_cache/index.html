private function get_pattern_cache() {
	if ( ! $this->exists() ) {
		return false;
	}

	$pattern_data = get_site_transient( 'wp_theme_files_patterns-' . $this->cache_hash );

	if ( is_array( $pattern_data ) && $pattern_data['version'] === $this->get( 'Version' ) ) {
		return $pattern_data['patterns'];
	}
	return false;
}
