public function enqueue() {
	parent::enqueue();

	$custom_background = get_theme_support( 'custom-background' );
	wp_localize_script(
		'customize-controls',
		'_wpCustomizeBackground',
		array(
			'defaults' => ! empty( $custom_background[0] ) ? $custom_background[0] : array(),
			'nonces'   => array(
				'add' => wp_create_nonce( 'background-add' ),
			),
		)
	);
}
