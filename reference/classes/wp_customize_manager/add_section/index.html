public function add_section( $id, $args = array() ) {
	if ( $id instanceof WP_Customize_Section ) {
		$section = $id;
	} else {
		$section = new WP_Customize_Section( $this, $id, $args );
	}

	$this->sections[ $section->id ] = $section;
	return $section;
}
