public function autoembed_callback( $matches ) {
	$oldval              = $this->linkifunknown;
	$this->linkifunknown = false;
	$return              = $this->shortcode( array(), $matches[2] );
	$this->linkifunknown = $oldval;

	return $matches[1] . $return . $matches[3];
}
