Source

protected function get_plugin_status( $plugin ) {
	if ( is_plugin_active_for_network( $plugin ) ) {
		return 'network-active';
	}

	if ( is_plugin_active( $plugin ) ) {
		return 'active';
	}

	return 'inactive';
}
